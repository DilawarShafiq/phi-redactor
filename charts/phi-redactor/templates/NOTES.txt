1. phi-redactor is now running. Get the proxy URL:

{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "phi-redactor.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "phi-redactor.fullname" . }} --template "{{"{{range (index .status.loadBalancer.ingress 0)}}{{.}}{{end}}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "phi-redactor.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:{{ .Values.config.port }}
  echo "Visit http://127.0.0.1:8080"
{{- end }}

2. Available proxy endpoints:

  OpenAI:    /openai/v1/chat/completions
  Anthropic: /anthropic/v1/messages
  Azure:     /azure/v1/chat/completions
  Google:    /google/v1beta/models/{model}:generateContent

  Health:    /api/v1/health
  Redact:    /api/v1/redact

3. Point your LLM client at the proxy instead of the provider:

  # Python openai SDK example:
  import openai
  client = openai.OpenAI(
      base_url="http://<proxy-url>/openai/v1",
      api_key="<your-openai-key>",
  )
